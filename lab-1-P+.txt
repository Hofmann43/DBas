
CREATE TABLE department(
department_name VARCHAR(255) PRIMARY KEY NOT NULL,
building_nr INT NOT NULL
);


CREATE TABLE employee(
employee_id INT PRIMARY KEY NOT NULL,
full_name VARCHAR(255) NOT NULL, 
phone_nr VARCHAR(255),
department_name VARCHAR(255) NOT NULL,
start_date DATE NOT NULL,
mentor_id INT,
CONSTRAINT fk_department
  FOREIGN KEY (department_name)
REFERENCES department(department_name),
  CONSTRAINT fk_mentor
FOREIGN KEY (mentor_id)
  REFERENCES employee (employee_id)
);


CREATE TABLE doctor(
doctor_id INT NOT NULL PRIMARY KEY,
specialization VARCHAR(255) NOT NULL,
room_nr INT NOT NULL,
CONSTRAINT fk_doctor
  FOREIGN KEY (doctor_id)
REFERENCES employee (employee_id)
);


CREATE TABLE nurse(
nurse_id INT NOT NULL PRIMARY KEY,
degree VARCHAR(255) NOT NULL
CONSTRAINT fk_nurse
  FOREIGN KEY (nurse_id)
REFERENCES employee (employee_id)
); 


CREATE TABLE patient(
patient_id INT PRIMARY KEY NOT NULL,
diagnosises VARCHAR(255),
age INT,
full_name VARCHAR(255) NOT NULL,
);


CREATE TABLE treating (
PRIMARY KEY (treating_dr_id,treating_pt_id),
treating_dr_id INT NOT NULL,
treating_pt_id INT NOT NULL,
CONSTRAINT fk_doctor
  FOREIGN KEY (treating_dr_id)
REFERENCES doctor (doctor_id),
  CONSTRAINT fk_patient
FOREIGN KEY (treating_pt_id)
  REFERENCES patient (patient_id)
);
